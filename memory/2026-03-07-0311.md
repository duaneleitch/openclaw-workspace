```python
import sys
import copy
from pptx import Presentation
from pptx.util import Inches, Pt
from pptx.enum.shapes import MSO_SHAPE
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN

# --- Brand Config ---
NAVY = RGBColor(0x23, 0x16, 0x8B)
TEAL = RGBColor(0x5D, 0xDE, 0xD2)
WHITE = RGBColor(0xFF, 0xFF, 0xFF)
GREY_TEXT = RGBColor(0x33, 0x33, 0x33)
FONT_NAME = 'Archivo'

def move_element_to_back(slide, shape):
    """Moves a shape to the bottom of the z-order (background)."""
    slide.shapes._spTree.remove(shape._element)
    slide.shapes._spTree.insert(2, shape._element) # Index 2 typically places it just above slide background

def apply_brand_layout(prs):
    slide_width = prs.slide_width
    slide_height = prs.slide_height
    
    # Layout Grid
    panel_width = int(slide_width * 0.35)
    footer_height = Inches(0.6)
    
    content_left_margin = panel_width + Inches(0.4)
    content_width = slide_width - content_left_margin - Inches(0.4)
    
    for slide in prs.slides:
        # Skip Title Slides (heuristic: usually Layout 0 or unique placeholders)
        if slide.slide_layout == prs.slide_layouts[0]:
            continue
            
        # 1. Add Visual Anchors (Navy "L" Frame)
        # Left Panel
        panel = slide.shapes.add_shape(
            MSO_SHAPE.RECTANGLE, 0, 0, panel_width, slide_height
        )
        panel.fill.solid()
        panel.fill.fore_color.rgb = NAVY
        panel.line.fill.background() # No outline
        move_element_to_back(slide, panel)
        
        # Footer Bar (Right side)
        footer = slide.shapes.add_shape(
            MSO_SHAPE.RECTANGLE, panel_width, slide_height - footer_height, 
            slide_width - panel_width, footer_height
        )
        footer.fill.solid()
        footer.fill.fore_color.rgb = NAVY
        footer.line.fill.background()
        move_element_to_back(slide, footer)

        # 2. Reflow Existing Content
        for shape in slide.shapes:
            if shape.has_text_frame:
                # If shape is in the "danger zone" (left panel), move it right
                if shape.left < panel_width:
                    # Maintain relative vertical position
                    shape.left = content_left_margin
                    
                    # Resize width to fit content area
                    if shape.width > content_width:
                        shape.width = content_width
                
                # Ensure Text Visibility & Font
                for paragraph in shape.text_frame.paragraphs:
                    paragraph.font.name = FONT_NAME
                    for run in paragraph.runs:
                        run.font.name = FONT_NAME
                        # Force dark text if not explicitly colored or if white
                        if not run.font.color.type or (run.font.color.rgb == WHITE):
                            run.font.color.rgb = GREY_TEXT

if __name__ == "__main__":
    pptx_path = sys.argv[1] if len(sys.argv) > 1 else "presentation.pptx"
    prs = Presentation(pptx_path)
    apply_brand_layout(prs)
    prs.save(pptx_path)
    print(f"Branded layout applied to {pptx_path}")
```
